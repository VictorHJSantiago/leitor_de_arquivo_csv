    private void jMenuAbrirActionPerformed(java.awt.event.ActionEvent evt) {                                           
        FileReader arquivo = null;
        String caminho = null;
        
        try (BufferedReader reader = new BufferedReader(new FileReader(caminhoArquivo))) {
            StringBuilder conteudo = new StringBuilder();
            String linha;
            while ((linha = reader.readLine()) != null) {
                conteudo.append(linha).append("\n");
            }
            textArea.setText(conteudo.toString());
        } catch (IOException e) {
            JOptionPane.showMessageDialog(this, "Erro ao ler o arquivo: " + e.getMessage(),
                    "Erro", JOptionPane.ERROR_MESSAGE);
        }
    }

    private void textField1ActionPerformed(java.awt.event.ActionEvent evt) {                                           
        String linha="";
        String divisorCSV=";";
        Boolean primeiraLinha = true;
        ArrayList<Venda> comercio = new ArrayList<>();
        try {
            BufferedReader br = new BufferedReader(new FileReader(caminhoArquivo));
            
            while ((linha = br.readLine()) != null) {
                if (!primeiraLinha) {
                    String[] colunas = linha.split(divisorCSV);

                    Venda local = new Venda();
                    local.setLoja(colunas[2]);
                    local.setPrecoUnitario(Double.parseDouble(colunas[6]));

                    Boolean existeLoja = false;
                    
                    for (int i=0; i<comercio.size(); i++) {
                        Venda item = comercio.get(i);
                        
                      
                        if (item.getLoja().equals(local.getLoja())) {
                           existeLoja = true;
                           item.setPrecoUnitario(local.getPrecoUnitario() + item.getPrecoUnitario());
                        }
                    }                           
                   
                    if (existeLoja == false)                    
                        comercio.add(local);
                    

                    if (comercio.size() == 0) {
                        comercio.add(local);
                    }

                }
                primeiraLinha = false;
            }

            comercio.forEach(item -> {
                System.out.println(item.getLoja() + "/" + item.getPrecoUnitario());
            });
            System.out.println(comercio.size());
        } catch (IOException e) {
            System.out.println("Erro: " + e.getMessage());
        }
    }
